import pickle
import os
import re

BASE_DIR = os.path.dirname(__file__)

# Load model
with open(os.path.join(BASE_DIR, "sqli_model.pkl"), "rb") as f:
    model = pickle.load(f)

with open(os.path.join(BASE_DIR, "sqli_vectorizer.pkl"), "rb") as f:
    vectorizer = pickle.load(f)


# 🔒 Rule-based detection (fast + reliable)
def rule_based_check(text):
    text = str(text).lower()

    patterns = [
        r"drop\s+table",
        r"delete\s+from",
        r"insert\s+into",
        r"update\s+.*set",
        r"alter\s+table",
        r"union\s+select",
        r"or\s+1=1",
        r"'--",
        r"--",
        r";--",
        r"/\*.*\*/"
    ]

    for pattern in patterns:
        if re.search(pattern, text):
            return True

    return False


# 🚀 Final prediction function
def predict_query(query_text, threshold=0.4):

    query_text = str(query_text)

    # 1️⃣ Rule-based check (highest priority)
    if rule_based_check(query_text):
        return True, 1.0

    # 2️⃣ ML prediction
    vector = vectorizer.transform([query_text])

    proba = model.predict_proba(vector)[0][1]  # probability of malicious

    is_malicious = proba >= threshold

    return is_malicious, float(proba)
